#### 예제 6-1. 학생들 성적을 입력받아 총점과 평균을 구하고, 성적순으로 출력하는 프로그램

```c
#include <stdio.h>
#define N 10

void input(int scr[], int size);
int sum2(int scr[], int size);
void sort(int scr[], int size);

void main(){
    int score[N] = {0};
    int j, sum;
    float average;

    input(score, N);
    sum = sum2(score, N);
    average = (float) sum/N;
    printf("평균 : %f\n", average);

    sort(score, N);
    printf("성적순 : ");
    for(j = 0; j < N; j++){
        printf("%d ", score[j]);
    }
    printf("\n");
}

void input(int scr[], int size){
    int j;
    for(j = 0; j < size; j++){
        printf("%d번째 성적을 입력하세요 : ", j+1);
        scanf("%d", &(scr[j]));
    }
}

int sum2(int scr[], int size){
    int sum, j;
    for(sum = 0, j = 0; j < size; j++){
        sum += scr[j];
    }
    return sum;
}

void sort(int scr[], int size){
    int j, k, temp;
    for(j = 0; j < size - 1; j++){
        for(k = size - 1; k > j; k--){
            if(scr[k-1] > scr[k]){
                temp = scr[k-1];
                scr[k-1] = scr[k];
                scr[k] = temp;
            }
        }
    }
}

/* 실행 결과

1번째 성적을 입력하세요 : 50
2번째 성적을 입력하세요: 66
3번째 성적을 입력하세요 : 59
4번째 성적을 입력하세요 : 90
5번째 성적을 입력하세요: 100
6번째 성적을 입력하세요: 98
7번째 성적을 입력하세요: 97
8번째 성적을 입력하세요: 50
9번째 성적을 입력하세요: 80
10번째 성적을 입력하세요 : 70
평균 : 76.000000
성적순 : 50 50 59 66 70 80 90 97 98 100
*/
```

#### 예제 6-2. 3*3 행렬 2개를 읽어 들여 합을 출력하는 프로그램
```c
#include <stdio.h>

void main(){
    int a[3][3] = {{9, 8, 7}, {6, 5, 4}, {3, 2, 1}};
    int b[3][3] = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
    int c[3][3];
    int i, j;

    for(i = 0; i < 3; i++){
        for(j = 0; j < 3; j++){
            c[i][j] = a[i][j] + b[i][j];
        }
    }

    printf("\n *** c = a + b *** \n");

    for(i = 0; i < 3; i++){
        for(j = 0; j < 3; j++){
            printf("%2d ", c[i][j]);
        }
        printf("\n");
    }
}
```
#### 예제 6-3. 3*3 정방행렬의 전치행렬을 구하는 프로그램
```c
#include <stdio.h>

void main(){
    int i, j, dim[3][3] = {{1, 2, 3}, {4, 5, 6}, {7, 8, 9}};
    const int max_cnt = sizeof(dim) / sizeof(dim[0]);

    printf("원래의 행렬 \n");
    for(i = 0; i < max_cnt; i++){
        for(j = 0; j < max_cnt; j++){
            printf("%d ", dim[i][j]);
        }
        printf("\n");
    }
    printf("\n");
    for(i = 0; i < max_cnt; i++){
        for(j = 0; j < i; j++){
            dim[i][j] ^= dim[j][i];
            dim[j][i] ^= dim[i][j];
            dim[i][j] ^= dim[j][i];
        }
    }
    printf("전치행렬 \n");
    for(i = 0; i < max_cnt; i++){
        for(j = 0; j < max_cnt; j++){
            printf("%d ", dim[i][j]);
        }
        printf("\n");
    }
}

/* 실행 결과

원래의 행렬 
1 2 3 
4 5 6 
7 8 9

전치행렬 
1 4 7 
2 5 8 
3 6 9
*/
```
- 배열
  - 동일 자료형의 값들이 순서적으로 하나의 이름(배열명)에 모여 있는 것
  - 각각의 자료들은 원소라 하며, 이들은 배열명과 첨자로 구분
  - 첨자의 표현은 대괄호({})로 나타냄
- 배열의 선언 형태
  - 1차원 배열 : 자료형 배열명[개수]
  - 2차원 배열 : 자료형 배열명[행의 수][열의 수]
  - 3차원 배열 : 자료형 배열명[면의 수][행의 수][열의 수]
  
#### N개의 정수를 읽어 들여 오름차순으로 정렬하는 프로그램
```c
#include <stdio.h>

void main(){
    static int data[100]; // 1차원 배열을 정적변수(static)로 선언
    int i, j, N;
    printf("입력할 데이터의 개수를 입력하시오 : ");
    scanf("%d", &N);
    for(i = 0; i < N; i++){
        scanf("%d", &data[i]);
    }
    for(i = 0; i < N-1; i++){
        for(j = i+1; j < N; j++){
            if(data[i] > data[j]){
                data[i] ^= data[j]; // XOR을 이용한 데이터 교환
                data[j] ^= data[i];
                data[i] ^= data[j];
            }
        }
    }
    printf("정렬된 데이터 : \n");
    for(i = 0; i < N; i++){
        printf("%d ", data[i]);    
    }
}

/* 실행 결과

입력할 데이터의 개수를 입력하시오 : 5
80
66
14
73
24
정렬된 데이터 : 
14 24 66 73 80
*/
```
#### 예제 6-5. 변수의 번지를 출력하는 프로그램
```c
#include <stdio.h>

void main(){
    int i = 127;
    int j;

    printf("%d, %x \n\n", i, i); // 변수 i의 값을 10진수와 16진수로 출력
    printf("%x \n", &i); // & 연산자 : 주소연산자 
    printf("%#x \n\n", &i); // #x : 16진수 출력시 0x를 먼저 표시(#0 : 8진수 출력시 0을 먼저 표시)
    printf("%x \n", &j);
    printf("%#x \n", &j);
}

/* 실행 결과

127, 7f 

1399bcd0 
0x1399bcd0 

1399bcd4 
0x1399bcd4
*/
```
#### 예제 6-6. 입력된 문자열을 역방향으로 출력하는 프로그램
```c
#include <stdio.h>
#include <conio.h>
#include <string.h>

void main(){
    char buffer[80], *p = buffer;
    int len, i = 0;
    printf("Input String: ");
    scanf("%s", p);
    len = strlen(p);
    while(*p){
        p++;
    }

    printf("Inverse String: ");
    while(i < len){
        p--; // 포인터의 위치를 하나씩 감소
        putchar(*p); // 감소된 포인터의 위치에 따라 역방향 출력
        i++;
    }

    putchar('\n');
    getch();
}

/* 실행 결과

Input String: apple
Inverse String: elppa
*/
```
